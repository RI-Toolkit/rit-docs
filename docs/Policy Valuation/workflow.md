# Workflow Example

```r
# Life Annuity
la <- create_policy_LA(60000, 5, 0.02, 0.05)
cf_la <- simulate_cf(la)
val_la <- value_policy(la, cf_la)

# Care Annuity
ca <- create_policy_CA(c(60000, 1200), c(0, 0.04), c(8, 0), c(0.04, 0.05))
cf_ca <- simulate_cf(ca)
val_ca <- value_policy(ca, cf_ca)
```
